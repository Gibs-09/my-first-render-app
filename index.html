const express = require("express"); const path = require("path"); const app = express(); const PORT = process.env.PORT || 3000; // IMPORTANT: correct static path app.use(express.static(path.join(__dirname, "public"))); // FORCE homepage app.get("/", (req, res) => { res.sendFile(path.join(__dirname, "public", "index.html")); }); // test route app.get("/a/:code", (req, res) => { res.send("Code: " + req.params.code); }); app.listen(PORT, () => { console.log("Server running on port " + PORT); });
